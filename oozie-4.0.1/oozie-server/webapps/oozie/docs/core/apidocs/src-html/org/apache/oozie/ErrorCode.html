241    private String template;
242    private int logMask;
243
244    /**
245     * Create an error code.
246     *
247     * @param template template for the exception message.
248     * @param logMask log mask for the exception.
249     */
250    private ErrorCode(int logMask, String template) {
251        this.logMask = logMask;
252        this.template = template;
253    }
254
255    /**
256     * Return the message (StringFormat) template for the error code.
257     *
258     * @return message template.
259     */
260    public String getTemplate() {
261        return template;
262    }
263
264    /**
265     * Return the log mask (to which log the exception should be logged) of the error.
266     *
267     * @return error log mask.
268     */
269    public int getLogMask() {
270        return logMask;
271    }
272
273    /**
274     * Return a templatized error message for the error code.
275     *
276     * @param args the parameters for the templatized message.
277     * @return error message.
278     */
279    public String format(Object... args) {
280        return XLog.format("{0}: {1}", toString(), XLog.format(getTemplate(), args));
281    }
